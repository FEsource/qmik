$.fn.un(name[,fun]),$.fn.unbind(name[,fun])

描述

取消绑定事件

使用方式

		方式1,清除对象的所有的click事件
		$("#bak").un("click");

		方式2,适用于清除对象上指定的click事件
		$("#bak").on("click",fun)

例子

<div id="bak">
	<div id="ban">
	</div>	
</div>
(function(Q){
	Q().ready(function(){
		function cclick(e){
			console.log("live click")
		}
		//绑定事件
		Q("#bak").on("click",cclick);
		
		//取消指定的click事件cclick
		Q("#bak").un("click",cclick);

		//取消所有的click事件
		Q("#bak").un("click");
	})
})(Qmik)
