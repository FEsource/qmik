$.fn.on(param),$.fn.bind(param)

描述

绑定事件

使用方式

		方式1,适合于定义单个事件
		$("#bak").on("click",function(e){console.log("test")});

		方式2,适用于定义多个事件
		$("#bak").on({
			click:function(e){
				console.log("test 1")
			},
			touchstart:function(e){
				console.log("test touch")
			}
		});
