$.sun.use(urls,callback)

描述

使用并加载相应的模块列表,可按功能划分模块,完成模块化开发,动态按需加载代码
使用:$.sun.use([url1,url2...],function(module1,module2...){})
参数说明:

例子

//Config.js配置文件
(function(Q){
	Q.sun.config({
		alias : {//别名系统(可以通过别名来引用js模块)
			// qmik组件
			"Qmik.nav" : Q.url("module/frame/qmik/Qmik.nav.js?${time}"),//
			"Qmik.fn.scrollBar" : Q.url("module/frame/qmik/Qmik.fn.scrollBar.js?${time}"),//
			// /
			"BaseAct" : "module/common/BaseAct.js?${time}",
			"QmikAPI" : "module/qmik/QmikAPI.js?${time}"
		},
		vars : {//变量系统
			//time:Q.now()
			time:function(){return parseInt(Q.now()/1000000)}
		}
	});
})(Qmik);

//index.html使用
//需要引用Qmik.js和Config.js文件
(function(Q){
	Q.sun.use("BaseAct",function(Base){
		var base=new Base();
	});
	//or
	Q.sun.use(["BaseAct","QmikAPI"],function(Base,QmikAPI){
		var base=new Base();
	});
})(Qmik)
