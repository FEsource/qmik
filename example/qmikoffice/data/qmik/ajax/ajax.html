fn.ajax(opts)

描述

发起ajax请求,支持跨域
opts参数格式 url:请求地址, data:请求参数 type : 'GET|POST', async : true|false, dataType : 'text|json|jsonp' //jsonp用于解决跨域的问题 timeout:10000//请求超时时间阀值 success:function(data){}//请求正常回调 error:function(){}//请求异常回调
