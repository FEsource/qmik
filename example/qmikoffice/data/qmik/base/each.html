each 方法

描述

循环对象
可以循环数组或类json型对象

例子

(function(Q){
	var ch={key:"me",age:11};
	Q.each(ch,function(key,val){
		console.log(key+":"+val);
	});
	//out:>>>
	/*
		key:me
		age:11
	*/

	ch=[1,2,3];
	Q.each(ch,function(key,val){
		console.log(key+":"+val);
	});
	//out:>>>
	/*
		0:1
		1:2
		2:3
	*/	
})(Qmik)
