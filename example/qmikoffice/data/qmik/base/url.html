url([url]) 方法

描述

生成新的url, 生成算法受config({context:"xxx"})配置条件的影响
生成规则
1.url()参数没传,返回当前的域名地址,格式为: protocol + "//" + hostname + (port ? ":" + port : "");如:http://www.qmik.org
2.url("http://www.qmik.org/xxxxx/xxx ")参数包含协议及域名时,不做处理,返回原值
3.url("/xxx/xxx")参数包含/开头,返回 protocol + "//" + hostname + (port ? ":" + port : "")+ "/xxx/xxx"
4.url("xxx/xxx")返回 protocol + "//" + hostname + (port ? ":" + port : "")+config.context+ "xxx/xxx"

例子

(function(Q){
	var url="";

	Q.config({
		context : "" 
	});
	console.log("config.context="+Q.config().context)
	//当前网址为:http://www.qmik.org
	console.log(url+">>"+Q.url());//>>>http://www.qmik.org
	url="index.html";
	console.log(url+">>"+Q.url(url));//>>>http://www.qmik.org/index.html
	url="/index.html";
	console.log(url+">>"+Q.url(url));//>>>http://www.qmik.org/index.html
	url="xx/index.html";
	console.log(url+">>"+Q.url(url));//>>>http://www.qmik.org/xx/index.html
	url="/xx/index.html";
	console.log(url+">>"+Q.url(url));//>>>http://www.qmik.org/xx/index.html
	console.log("-------------------------------------------------");
	

	Q.config({
		context : "/qmik/q1/" 
	});
	console.log("config.context="+Q.config().context)
	url="";
	//当前网址为:http://www.qmik.org
	console.log(url+">>"+Q.url());//>>>http://www.qmik.org
	url="index.html";
	console.log(url+">>"+Q.url(url));//>>>http://www.qmik.org/index.html
	url="/index.html";
	console.log(url+">>"+Q.url(url));//>>>http://www.qmik.org/index.html
	url="xx/index.html";
	console.log(url+">>"+Q.url(url));//>>>http://www.qmik.org/xx/index.html
	url="/xx/index.html";
	console.log(url+">>"+Q.url(url));//>>>http://www.qmik.org/xx/index.html
	console.log("-------------------------------------------------");
})(Qmik)
