ajax({}) 方法

描述

发起ajax请求,请求远程数据,如果是跨域的请求,请走dataType=jsonp类型
url:"/url" //请求地址 data:{}//请求参数 dataType:"json|text|jsonp" //返回类型 async:true|false//是否走异步,默认是,对jsonp无效 timeout:1000,//超时时间,单位毫秒 success:function(data){}//成功返回数据 error:function(){}//请求失败

例子

(function(Q){
 Q.ajax({
  url:"/",
  data:{},
  dataType:"json",
  timeout:15000,
  async:true,
  success:function(data){

  },
  error:function(data){

  }
 });
})(Qmik)
