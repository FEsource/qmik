merge(arr1,...) 方法

描述

把多个对象合并为一个对象返回,(合并到第一个对象上去)
如果第一个参数是数组,返回合并后的数据,
如果第一个参数是{}对象,返回合并后的对象

例子

(function(Q){
	var arr1=[1];
	var arr2=[3,6];
	var obj1={key:"me",age:11};
	var rArr=Q.merge(arr1,arr2,obj1);
	console.log(rArr);//>>>:[1,3, 6, "me", 11]

	var arr1=[1];
	var arr2=[3,6];
	var obj1={key:"me",age:11};
	var rArr=Q.merge(obj1,arr1,arr2);
	console.log(rArr);//>>>:Object {0: 3, 1: 6, key: "me", age: 11}
})(Qmik)
