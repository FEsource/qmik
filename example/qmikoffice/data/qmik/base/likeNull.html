likeNull 方法

描述

判断对象是否像一个空指针,或未定义的变量,或空值等
字符串 "","null","undefined"等也返回true

例子

(function(Q){
	var ch=null;
	var ch1=Q.likeNull(ch);
	console.log(ch1);//out:>>>true

	delete ch;
	ch1=Q.likeNull(ch);
	console.log(ch1);//out:>>>true

	ch="";
	ch1=Q.likeNull(ch);
	console.log(ch1);//out:>>>true

	ch="null";
	ch1=Q.likeNull(ch);
	console.log(ch1);//out:>>>true

	ch="undefined";
	ch1=Q.likeNull(ch);
	console.log(ch1);//out:>>>true

})(Qmik)
