Qmik模块开发快速入门(Qmik.sun)

define,定义模块 
    // User.js用户模块,模块根地址: /module/user/User.js
    (function($,define){//此头可以去掉,建议增加
        define(function(require,exports,module){
           $.extend(exports,{
                //显示用户列表
                showUserList:function(){
                    $.log("showUserList...")
                },
                //显示用户详细信息
                showUserInfo:function(userId){
                    $.log("showUserInfo..."+userId)
                }
            });
        });
    })(Qmik,Qmik.define);
    
    // Session.js 会话模块,模块根地址: /module/session/Session.js
    (function($,define){//此头可以去掉,建议增加
        define(function(require,exports,module){
           $.extend(exports,{
                //用户登录
                login:function(userId){
                    $.log("login..."+userId)
                },
                //用户注销
                logout:function(userId){
                    $.log("logout..."+userId)
                }
            });
        });
    })(Qmik,Qmik.define);
配置$.sun.config 对$.sun进行配置 
    $.sun.config({
        // 设置别名，方便调用,使用模块时,可以使用 别名或直接地址
        alias: {
            "User" : "p_user/User",
            "Session" : "/module/session/Session",
        }
    });
    $.sun.config({
        // 设置路径，方便跨目录调用,路径中的 p_user会替换成/module/user
        paths: {
            "p_user" : "/module/user"
        }
    });
require,exports,module 说明 
	require:局部方法,需要请求加载的模块,同步执行,
	     例如
	     define(function(require,exports,module){
	        var user=require("User");
	        user.showUserList();
	     });
	     
	 exports:是module.exports的引用,用于导航对象
	 
	 module:模块实例,源自Module的实例,包含id,url,exports等属性
配置$.use,$.sun.use 加载一个或多个模块 
	//加载一个模块
     $.use("User");或 $.use("/module/user/User");
     
      //加载一个模块并使用
      $.use("User",function(user){
         user.showUserList();
      })
      
      //加载多个模块
      $.use(["User","Session"]);或 $.use(["/module/user/User","/module/session/Session"]);
     
      //加载多个模块并使用
      $.use(["User","Session"],function(user,session){
         user.showUserList();
         session.login(1);
      })
